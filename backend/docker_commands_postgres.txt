
Run / downloads postgres:alpine image and starts the container to run the application

docker run -d \
	--name postgres \
	-p 6000:5432 \
	-e POSTGRES_PASSWORD=coyotlPsql2022! \
	// Comment: here you have to define the path to where you want to store the data
	-v /home/yourname/development/database/postgres:/var/lib/postgresql/data \
	postgres:alpine
